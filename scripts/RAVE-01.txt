on *:CTCPREPLY:VERSION*: { .echo -sant 14*4 $nick 14is using IRC client:4 $2- | halt }
on *:CTCPREPLY:PING*: {
  if ((%rave.ping. [ $+ [ $nick ] ] == 1) && ($comchan($nick,1) != $null)) {
    .notice $nick  $+ $nick $+ : Your PING Reply is ( $duration($calc(($ticks - $2)/1000)) ) 5-[ 4RAVE v4.01 5]-
    .echo -tn @intel  $+ $color(notice) $+ *** $nick $+ 's PING Reply is $duration($calc(($ticks - $2)/1000)) (Reply sent to $nick $+ )
    .unset %rave.ping. [ $+ [ $nick ] ]
  }
  else { echo -st 14*4 $nick 14Ping $calc($ticks - $2) Reply:4 $duration($calc(($ticks - $2)/1000)) | inc %activecount }
  haltdef
}
on *:TEXT:*:#: {
  if ((%rave.pingswitch == OFF) || (%rave.ping.pause == ON)) { goto end }
  if (!ping isin $strip($1-)) {
    set -u5 %rave.ping.pause ON
    set %rave.ping. [ $+ [ $nick ] ] 1
    .ctcp $nick ping 
  } 
  :end
}
on *:ACTION:*:#: {
  if ((%rave.pingswitch == OFF) || (%rave.ping.pause == ON)) { goto end }


  if (!ping isin $strip($1-)) {
    set -u5 %rave.ping.pause ON
    set %rave.ping. [ $+ [ $nick ] ] 1
    .ctcp $nick ping 
  } 
  :end
}

alias ak.addchan {
  if ($left($1,1) == $null) {
    .echo -q $input(Please enter a channel name.,68,Error)
    halt
  }
  if ($left($1,1) != $chr(35)) {
    if ($chr(44) !isin $1) {
      if ($istok(%mega.mch,$chr(35) $+ $1,32) == $true) { .echo -q $input(Channel already exists!,68,Error) | halt }
      set %mega.mch %mega.mch $chr(35) $+ $1
      set %chan.set.def $chr(35) $+ $1
    }
    goto done
  }
  if ($left($1,1) == $chr(35)) {
    if ($chr(44) !isin $1) {
      if ($istok(%mega.mch,$1,32) == $true) { .echo -q $input(Channel already exists!,68,Error) | halt }
      set %mega.mch %mega.mch $1
      set %chan.set.def $1
    }
    goto done
  }
  .echo -q $input(Please enter a valid channel name.,68,Error)
  halt
  :done
  set %set.default.pars $input(Would you like to use the default $crlf $+ protection settings for %chan.set.def $+ ?,y, %chan.set.def)
  ak.set.defaults %chan.set.def
  unset %chan.set.def
  :end
}
